 36class ObjectSelection(Gtk.Frame):
 37    def __init__(self, parent, exclude_list):
 38        super().__init__()
 39        self.parent = parent
 40        self.exclude_list = exclude_list
 41        labels_list = self.exclude_list.copy()
 42        labels_list.sort()
 43
 44        vertical_box = Gtk.Box(orientation=Gtk.Orientation.VERTICAL)
 45        self.add(vertical_box)
 46
 47        scrolled_window = Gtk.ScrolledWindow()
 48
 49        horizontal_box = Gtk.Box(orientation=Gtk.Orientation.HORIZONTAL)
 50        quit_button = Gtk.Button.new_with_label('Quit')
 51        quit_button.connect('clicked', self.on_quit_button_clicked)
 52        back_button = Gtk.Button.new_with_label('Back')
 53        back_button.connect('clicked', self.on_back_button_clicked)
 54        horizontal_box.pack_start(back_button, True, True, 10)
 55        horizontal_box.pack_start(quit_button, True, True, 10)
 56        vertical_box.pack_start(horizontal_box, False, True, 10)
 57        vertical_box.pack_start(scrolled_window, True, True, 10)
 58
 59        vertical_box = Gtk.Box(orientation=Gtk.Orientation.VERTICAL)
 60        scrolled_window.add(vertical_box)
 61
 62        for label in labels_list:
 63            horizontal_box =  Gtk.Box(orientation=Gtk.Orientation.HORIZONTAL)
 64            switch_button = Gtk.Switch()
 65            switch_button.set_active(False)
 66            switch_button.connect('notify::active',
 67                                  self.on_object_switch_activated, label)
 68            label_name = Gtk.Label.new(label)
 69            horizontal_box.pack_start(label_name, True, True, 100)
 70            horizontal_box.pack_start(switch_button, False, True, 100)
 71            vertical_box.pack_start(horizontal_box, True, True, 10)

 73    def on_quit_button_clicked(self, button):
 74        Gtk.main_quit()
 75
 76    def on_back_button_clicked(self, button):
 77        self.parent.set_current_page(0)
 78
 79    def on_object_switch_activated(self, switch, gparam, obj):
 80        if switch.get_active():
 81            if obj in self.exclude_list:
 82                self.exclude_list.remove(obj)
 83        else:
 84            if obj not in self.exclude_list:
 85                self.exclude_list.append(obj)

157    def set_displayed_image(self, image):
158        image = cv2.resize(image, (420, 340))
159        image = cv2.cvtColor(image, cv2.COLOR_RGB2BGR)
160        height, width = image.shape[:2]
161        arr = np.ndarray.tobytes(image)
162        self.pixbuf = Pixbuf.new_from_data(
163                             arr, Colorspace.RGB, False, 8,
164                             width, height, width*3, None, None)
165        self.displayed_image.set_from_pixbuf(self.pixbuf)
166        self.pixbuf = None
167
168    def run_application(self):
169        thread = threading.Thread(target=self.image_detection)
170        thread.daemon = True
171        thread.start()
172
173    def start_interpreter(self):
174        self.engine = TFLiteInterpreter(self.model_file_path)
175
176    def image_detection(self):
177        resizer = Resizer()
178        resizer.set_sizes(engine_input_details=self.engine.input_details)
179
180        camera = Multimedia("/dev/video1", resolution="vga")
181        camera.set_v4l2_config()
182        
183        framerate = Framerate()
184
185        draw = Overlay()
186        draw.inference_time_info = False
187        draw.scores_info = True
188        draw.extra_info = False
189        draw.framerate_info = False
190        
191        while camera.loop:
192            with framerate.fpsit():
193                frame = camera.get_frame()
194                resizer.resize_frame(frame)
195
196                self.engine.set_input(resizer.frame_resized)
197                self.engine.run_inference()
198
199                positions = self.engine.get_output(0, squeeze=True)
200                classes = self.engine.get_output(1, squeeze=True)
201                scores = self.engine.get_output(2, squeeze=True)
202                
203                result = []
204                for idx, score in enumerate(scores):
205                    if score > 0.5 and (self.labels[classes[idx]] not in self.exclude_list):
206                        result.append({'pos': positions[idx], '_id': classes[idx]})
207                
208                output_frame = draw.info(category="detection",
209                                         image=resizer.frame,
210                                         top_result=result,
211                                         labels=self.labels,
212                                         inference_time=None,
213                                         model_name=None,
214                                         source_file=camera.dev.name,
215                                         fps=None)
216
217            GLib.idle_add(self.inference_value_label.set_text,
218                          (f"{self.engine.inference_time}"))
219            GLib.idle_add(self.fps_value_label.set_text,
220                          (f"{int(framerate.fps)}"))
221            GLib.idle_add(self.set_displayed_image, (output_frame))
